SET(
  NAME
  arm_debugger
)

INCLUDE_DIRECTORIES(
  ${CMAKE_CURRENT_BINARY_DIR}
)

FILE( GLOB SRCS *.cpp )
FILE( GLOB HEADER *.h )

SET( MOC_SRCS )

MY_WRAP_CPP(
	MOC_SRCS
	"${arm_debugger}"
	${HEADER}
)

ADD_EXECUTABLE(
	${NAME}
	WIN32
	${SRCS}
	${HEADER}
	${MOC_SRCS}
)

IF(WIN32 AND NOT MINGW)
	SET_TARGET_PROPERTIES( ${NAME} PROPERTIES COMPILE_FLAGS "/MP" )
ENDIF(WIN32 AND NOT MINGW)

TARGET_LINK_LIBRARIES(
	${NAME}
	${QT_QTCORE_LIBRARY}
	${QT_QTMAIN_LIBRARY}
	${QT_QTNETWORK_LIBRARY}
	${QT_QTXML_LIBRARY}
	${QT_QTGUI_LIBRARY}
	${QT_HELP_LIBRARIES}
	rec_serialport
	rec_grappler
)

SOURCE_GROUP( "Source Files" FILES ${SRCS} ${HEADER} )
SOURCE_GROUP( Moc REGULAR_EXPRESSION "(moc_|qrc_)" )

#GENERAT_VERSION( ${NAME} "${CMAKE_CURRENT_BINARY_DIR}/debugger_version.h" ${ROBOTINO_ARM_VERSION} "grappler" "robotino_arm_debugger.exe" )

INSTALL(
  TARGETS 
  ${NAME}
  DESTINATION ${BIN_INSTALL_DIR}
)

FILE( GLOB INF ${CMAKE_CURRENT_SOURCE_DIR}/*.inf )
	
INSTALL(
	FILES
	${INF}
	DESTINATION ${BIN_INSTALL_DIR}
)
